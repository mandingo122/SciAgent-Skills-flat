\section{Study 1}

\begin{frame}{Study 1: Design and Methods}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{Design:} Cross-sectional survey\\
      \textbf{N} = 500, online recruitment\\
      \textbf{Power:} .95 for medium effects\\[0.3cm]
      \textbf{Hypotheses:}
      \begin{enumerate}
        \item H1: X positively predicts Y
        \item H2: Z moderates X $\to$ Y
      \end{enumerate}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{Measures:}
      \begin{itemize}
        \item X: Validated scale ($\alpha = .89$)
        \item Y: Performance measure
        \item Z: Individual difference
      \end{itemize}
      \textbf{Analysis:} Hierarchical regression, PROCESS
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Study 1: Results}
  \begin{columns}[c]
    \begin{column}{0.6\textwidth}
      % \includegraphics[width=\textwidth]{study1.pdf}
      \framebox[0.9\textwidth][c]{[Study 1 Main Result]}
    \end{column}
    \begin{column}{0.4\textwidth}
      \textbf{Findings:}
      \begin{itemize}
        \item H1 supported ($\beta = .47$, $R^2 = .22$)
        \item H2 supported (moderation by Z)
      \end{itemize}
      \begin{block}{Conclusion}
        Establishes foundational X $\to$ Y relationship.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\section{Study 2}

\begin{frame}{Study 2: Design and Methods}
  \textbf{Design:} Three-wave longitudinal ($n=250$, 6 months apart)\\
  \textbf{Measures:} X (T1), A and B (T2), Y (T3)\\
  \textbf{Analysis:} SEM, cross-lagged panel, FIML for missing data\\[0.5cm]
  \textbf{Retention:} 88\% at T2, 82\% at T3
\end{frame}

\begin{frame}{Study 2: Mediation Results}
  \begin{figure}

    \framebox[0.7\textwidth][c]{[Serial Mediation: X $\to$ A $\to$ B $\to$ Y]}

  \begin{itemize}
    \item Indirect effect: $\beta = 0.29$, 95\% CI [0.19, 0.41]
    \item \alert{61\% of total effect mediated by A $\to$ B pathway}
  \end{itemize}
\end{frame}
